def J_theta(X, y, theta):
    epsilon = 1e-6
    h = h_theta(X, theta)
    N = X.shape[0]
    tmp = y * np.log(h + epsilon)+(1 - y)*np.log(1 - h + epsilon)
    return np.sum(tmp) / N

def gradientAscend(X, y, learning_rate, num_epoch):
    N, D = X.shape
    theta = np.zeros(D)
    J = []
    
    for i in range(num_epoch):
        for t in range(D):
            cost = 0
            for i in range(N):
                cost += -J_theta(X, y, theta)
            cost *= learning_rate * 1.0 / N
            theta[t] -= cost
        J.append(cost)
    return theta, J
